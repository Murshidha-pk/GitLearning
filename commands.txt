git config--global user.name "user.name"   =>[for user name]

git config--global user.email "user.email"  =>[for git email]

git init  =>[initialise git in current working directory]

git status  =>[list untracked files]

git add -A   =>[to add all untracked files to staging area]

git commit -m   =>[to commit with descriptive comit meaasage]

git remote add remote_name path   =>[for adding remote repository]
eg: git remote add origin https://github.com/Murshidha-pk/GitLearning.git

git push remote_name branch
eg: git push origin master

git branch   =>[list all branches]
